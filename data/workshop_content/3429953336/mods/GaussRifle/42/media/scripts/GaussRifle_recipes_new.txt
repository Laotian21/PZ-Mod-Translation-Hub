	{

		Time = 1,
		category = Weaponry,
        Tags = InHandCraft,
		inputs
		{
			item 1 [Base.x2Scope;Base.x4Scope;Base.x8Scope;Base.RedDot] mappers[attachments],
		}
		outputs
		{
			item 1 mapper:attachments,
		}
        	itemMapper attachments
        {
			GaussRifle.Gaussx2Scope = Base.x2Scope,
			GaussRifle.Gaussx4Scope = Base.x4Scope,
			GaussRifle.Gaussx8Scope = Base.x8Scope,

			default = GaussRifle.Gaussx2Scope,
		}
