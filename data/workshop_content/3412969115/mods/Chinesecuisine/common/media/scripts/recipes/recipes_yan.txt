	craftRecipe Open box of Chinese cigarettes	/* 打开一盒中华香烟 */
	{
        timedAction = Making,
		Time = 30,
        OnTest = Recipe.OnTest.IsFull,
		/* CanWalk = true, */
        Tags = InHandCraft;CanBeDoneInDark,
        category = Miscellaneous,
		inputs
		{
			item 1 [Chinesecuisine.Chinese_CigarettePack] mode:destroy flags[IsFull;AllowFavorite;InheritFavorite],
		}
		outputs
		{
			item 20 Chinesecuisine.Chinese_CigaretteSingle,
		}
	}
